
d_emex_sw_machine_wasteland = {
	
	time = 180
	icon = "d_organic_landfill"
	is_for_colonizable = yes
	category = deposit_cat_blockers
	potential = { always = no }
	resources = {
		category = deposit_blockers
		cost = {
			energy = 2000
		}
	}
	
	drop_weight = {
		weight = 0
	}
	
	on_cleared = {
		hidden_effect = {
			if = {
				limit = {
					any_owned_pop = {
						is_robot_pop = yes
					}
				}
				random_owned_pop = {
					limit = {
						is_robot_pop = yes
					}
					root = {
						create_pop = {
							species = prev
						}
					}
				}
			}
		}
		custom_tooltip = d_decrepit_dwellings_custom_tooltip
	}
	
	planet_modifier = {
		planet_max_districts_add = -1
	}
}
